func get_input_map() -> Dictionary[UserInput, Array]:
	var data: Dictionary[UserInput, Array]
	
	for key in data.keys():
		var input := data[key]
		if data.has(key):
			data[key].append(input)
		else:
			data.set(key, Array())
	
	return data
