pragma solidity >=0.8.30;

  

enum Status {

    Waiting,

    Primaries,

    General,

    Electoral,

    Canceled,

    Finished

}

  

contract ElectoralCollege {

    address[] public chairPeople;

    address[] public officialPeople;

    Status public status = Status.Waiting;

  

    // Constructor.

    constructor(address initialChairPerson) {

        chairPeople.push(initialChairPerson);

    }

    // Reserved for only ChairPerson.

    function addChairPerson(address _address) public chairPersonOnly {

        chairPeople.push(_address);

    }

    // Reserved for only ChairPerson.

    function addOfficialPerson(address _address) public chairPersonOnly {

        officialPeople.push(_address);

    }

    // Reserved for only ChairPerson.

    function setStatus(Status _status) public chairPersonOnly {

        status = _status;

  

        if (status == Status.Waiting) {} else if (

            status == Status.Primaries

        ) {} else if (status == Status.General) {} else if (

            status == Status.Electoral

        ) {} else if (status == Status.Canceled) {} else if (

            status == Status.Finished

        ) {}

    }

  

    modifier chairPersonOnly() {

        bool existed = false;

  

        for (uint index = 0; index < chairPeople.length; index++) {

            if (chairPeople[index] == msg.sender) {

                existed = true;

                break;

            }

        }

  

        require(existed, "The caller isn't a chair person.");

  

        _;

    }

    modifier officialPersonOnly() {

        bool existed = false;

  

        for (uint index = 0; index < officialPeople.length; index++) {

            if (officialPeople[index] == msg.sender) {

                existed = true;

                break;

            }

        }

  

        require(existed, "The caller isn't a official person.");

  

        _;

    }

}
