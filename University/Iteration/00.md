  

enum Status {

    Waiting,

    Primaries,

    General,

    Electoral,

    Canceled,

    Finished

}

  

struct Party {

    uint8 partyID;

    uint64[] members;

}

  

contract ElectoralCollege {

    // List of chairpeople for moderating and managing the contract.

    address[] chairPeople;

    // List of officals for prodiving the local ability to vote.

    address[] officialPeople;

    //

    Status status = Status.Waiting;

    //

    Party[] parties;

    //

    uint8[] revokedParties;

  

    //

    uint64[] partyCandidatesVotes;

    //

    mapping(uint64 => uint64) candidateVotes;

    //

    mapping(uint8 => uint64) presidentCandidates;

  

    //

    uint64[] citizenVotes;

    //

    mapping(uint64 => uint8) partyVotes;

    // Sets the current status.

    function setStatus(Status _status) isChairPeople public {

        status = _status;

    }

    // Adds a new party to the list.

    function addParty(uint64[] memory members) isChairPeople public {

        require(parties.length < 256, "Party amount exceeds limits");

        parties.push(Party(uint8(parties.length), members));

    }

    // Revokes the party from the list.

    function revokeParty(uint8 partyID) isChairPeople public {

        revokedParties.push(partyID);

    }

  

    // Vote for candidate within the party.

    function voteCandidate(uint64 socialSecurityNumber, uint64 forCandidate) isOfficialPeople public {

    }

    // Vote for the party. Only allowed for the official people to uses this function.

    function voteParty(uint64 socialSecurityNumber, uint8 partyID) isOfficialPeople isQualifiedParty(partyID) public {

        for (uint256 index = 0; index < citizenVotes.length; index++) {

            if (citizenVotes[index] == socialSecurityNumber) {

                require(false, "This citizen has already voted.");

            }

        }

        citizenVotes.push(socialSecurityNumber);

        partyVotes[socialSecurityNumber] = partyID;

    }

  

    // Checks if the the person is a chairperson.

    modifier isChairPeople() {

        bool existed = false;

        for (uint256 x = 0; x < chairPeople.length; x++) {

            if (chairPeople[x] == msg.sender) {

                existed = true;

                break;

            }

        }

        require(existed, "You must be a chairpeople to do this.");

  

        _;

    }

    // Checks if the the person is a officalperson.

    modifier isOfficialPeople() {

        bool existed = false;

        for (uint256 x = 0; x < officialPeople.length; x++) {

            if (officialPeople[x] == msg.sender) {

                existed = true;

                break;

            }

        }

        require(existed, "You must be a chairpeople to do this.");

  

        _;

    }

    // Checks if the the person voting is inside the party.

    modifier isPartyMember(uint64 socialSecurityNumber, uint8 partyID) {

        bool existed = false;

        uint256 partyIndex = 0;

        // Checks for the index of the partyID

        for (uint256 index = 0; index < parties.length; index++) {

            if (parties[index].partyID == partyID) {

                partyIndex = index;

                break;

            }

        }

        // Checks if the person with socialSecurityNumber existed within the party.

        for (uint256 index = 0; index < parties[partyIndex].members.length; index++) {

            if (parties[partyIndex].members[index] == socialSecurityNumber) {

                existed = true;

                break;

            }

        }

        require(existed, "You must be a a party member to do this.");

  

        _;

    }

    // Checks if the partyID has not been revoked.

    modifier isQualifiedParty(uint8 partyID) {

        for (uint256 index = 0; index < revokedParties.length; index++) {

            require(

                revokedParties[index] == partyID,

                "The party has been revoked."

            );

        }

        _;

    }

}
